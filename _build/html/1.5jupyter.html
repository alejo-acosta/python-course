import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
sns.set_theme()

%matplotlib inline

n = 1000
c = np.random.choice([1,2],n)
x = np.random.normal(0,1,n) + c*2
y = np.random.normal(0,1,n) + c*1.5

array = np.array([x,y,c])

array.shape

(3, 1000)

df = pd.DataFrame(array, index=['x','y','c']).T
